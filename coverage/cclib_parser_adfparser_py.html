37 def normalisesym(self, label): 

38 """Use standard symmetry labels instead of ADF labels. 

39 

40 To normalise: 

41 (1) any periods are removed (except in the case of greek letters) 

42 (2) XXX is replaced by X, and a " added. 

43 (3) XX is replaced by X, and a ' added. 

44 (4) The greek letters Sigma, Pi, Delta and Phi are replaced by 

45 their lowercase equivalent. 

46 """ 

47 greeks = ['Sigma', 'Pi', 'Delta', 'Phi'] 

48 for greek in greeks: 

49 if label.startswith(greek): 

50 return label.lower() 

51 

52 ans = label.replace(".", "") 

53 if ans[1:3] == "''": 

54 temp = ans[0] + '"' 

55 ans = temp 

56 

57 l = len(ans) 

58 if l > 1 and ans[0] == ans[1]: # Python only tests the second condition if the first is true 

59 if l > 2 and ans[1] == ans[2]: 

60 ans = ans.replace(ans[0]*3, ans[0]) + '"' 

61 else: 

62 ans = ans.replace(ans[0]*2, ans[0]) + "'" 

63 return ans

65 def normalisedegenerates(self, label, num, ndict=None): 

66 """Generate a string used for matching degenerate orbital labels 

67 

68 To normalise: 

69 (1) if label is E or T, return label:num 

70 (2) if label is P or D, look up in dict, and return answer 

71 """ 

72 

73 if not ndict: 

74 ndict = { 

75 'P': {0: "P:x", 1: "P:y", 2: "P:z"}, 

76 'D': {0: "D:z2", 1: "D:x2-y2", 2: "D:xy", 3: "D:xz", 4: "D:yz"} 

77 } 

78 

79 if label in ndict: 

80 if num in ndict[label]: 

81 return ndict[label][num] 

82 else: 

83 return "%s:%i" % (label, num+1) 

84 else: 

85 return "%s:%i" % (label, num+1)
