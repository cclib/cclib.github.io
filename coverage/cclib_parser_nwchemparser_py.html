1091 def after_parsing(self): 

1092 """NWChem-specific routines for after parsing a file. 

1093 

1094 Currently, expands self.shells() into self.aonames. 

1095 """ 

1096 

1097 # setup a few necessary things, including a regular expression 

1098 # for matching the shells 

1099 table = utils.PeriodicTable() 

1100 elements = [table.element[x] for x in self.atomnos] 

1101 pattern = re.compile("(\ds)+(\dp)*(\dd)*(\df)*(\dg)*") 

1102 

1103 labels = {} 

1104 labels['s'] = ["%iS"] 

1105 labels['p'] = ["%iPX", "%iPY", "%iPZ"] 

1106 if self.shells['type'] == 'spherical': 

1107 labels['d'] = ['%iD-2', '%iD-1', '%iD0', '%iD1', '%iD2'] 

1108 labels['f'] = ['%iF-3', '%iF-2', '%iF-1', '%iF0', 

1109 '%iF1', '%iF2', '%iF3'] 

1110 labels['g'] = ['%iG-4', '%iG-3', '%iG-2', '%iG-1', '%iG0', 

1111 '%iG1', '%iG2', '%iG3', '%iG4'] 

1112 elif self.shells['type'] == 'cartesian': 

1113 labels['d'] = ['%iDXX', '%iDXY', '%iDXZ', 

1114 '%iDYY', '%iDYZ', 

1115 '%iDZZ'] 

1116 labels['f'] = ['%iFXXX', '%iFXXY', '%iFXXZ', 

1117 '%iFXYY', '%iFXYZ', '%iFXZZ', 

1118 '%iFYYY', '%iFYYZ', '%iFYZZ', 

1119 '%iFZZZ'] 

1120 labels['g'] = ['%iGXXXX', '%iGXXXY', '%iGXXXZ', 

1121 '%iGXXYY', '%iGXXYZ', '%iGXXZZ', 

1122 '%iGXYYY', '%iGXYYZ', '%iGXYZZ', 

1123 '%iGXZZZ', '%iGYYYY', '%iGYYYZ', 

1124 '%iGYYZZ', '%iGYZZZ', '%iGZZZZ'] 

1125 else: 

1126 self.logger.warning("Found a non-standard aoname representation type.") 

1127 return 

1128 

1129 # now actually build aonames 

1130 # involves expanding 2s1p into appropriate types 

1131 

1132 self.aonames = [] 

1133 for i, element in enumerate(elements): 

1134 try: 

1135 shell_text = self.shells[element] 

1136 except KeyError: 

1137 del self.aonames 

1138 msg = "Cannot determine aonames for at least one atom." 

1139 self.logger.warning(msg) 

1140 break 

1141 

1142 prefix = "%s%i_" % (element, i + 1) # (e.g. C1_) 

1143 

1144 matches = pattern.match(shell_text) 

1145 for j, group in enumerate(matches.groups()): 

1146 if group is None:
