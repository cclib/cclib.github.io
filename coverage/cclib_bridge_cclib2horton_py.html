11import numpy 

12from cclib.parser.data import ccData 

13from cclib.parser.utils import find_package

15# First check horton version 

16_old_horton = False 

17_found_horton = find_package("horton") 

18_found_iodata = find_package("iodata") 

19 

20# Detect whether horton 2 is present or not 

21if _found_horton: 

22 # Older horton packages do not have __version__, causing exceptions. 

23 try: 

24 from horton import __version__ 

25 except: 

26 _old_horton = True 

27 else: 

28 if __version__[0] == "2": 

29 from horton.io.iodata import IOData

32# Horton 3 is divided into smaller (sub)packages that each take different functionalities. 

33if _found_iodata: 

34 from iodata import IOData

38def check_horton(): 

39 if _old_horton: 

40 raise ImportError("You must have at least version 2 of `horton` to use this function.") 

41 elif not _found_horton and not _found_iodata: 

42 raise ImportError("You must install `horton` to use this function.") 

43 if _found_iodata: 

44 return 3 

45 elif _found_horton: 

46 return 2

49def makehorton(data): 

50 """ Create horton IOData object from ccData object """ 

51 

52 hortonver = check_horton() 

53 attributes = {} 

54 

55 if hortonver == 2: 

56 renameAttrs = {"atomnos": "numbers", "mult": "ms2", "polarizability": "polar"} 

57 inputattrs = data.__dict__ 

58 

59 attributes = dict( 

60 (renameAttrs[oldKey], val) 

61 for (oldKey, val) in inputattrs.items() 

62 if (oldKey in renameAttrs) 

63 ) 

64 

65 # Rest of attributes need some manipulation in data structure. 

66 if hasattr(data, "atomcoords"): 

67 # cclib parses the whole history of coordinates in the list, horton keeps the last one. 

68 attributes["coordinates"] = data.atomcoords[-1] 

69 if hasattr(data, "mocoeffs"): 

70 attributes["orb_alpha"] = data.mocoeffs[0] 

71 if len(data.mocoeffs) == 2: 

72 attributes["orb_beta"] = data.mocoeffs[1] 

73 if hasattr(data, "coreelectrons"): 

74 # cclib stores num of electrons screened out by pseudopotential 

75 # horton stores num of electrons after applying pseudopotential 

76 attributes["pseudo_numbers"] = data.atomnos - numpy.asanyarray(data.coreelectrons) 

77 if hasattr(data, "atomcharges") and ("mulliken" in data.atomcharges): 

78 attributes["mulliken_charges"] = data.atomcharges["mulliken"] 

79 if hasattr(data, "atomcharges") and ("natural" in data.atomcharges): 

80 attributes["npa_charges"] = data.atomcharges["natural"] 

81 

82 elif hortonver == 3: 

83 # In horton 3 IOData, inputs are type-checked -- thus the bridge also verifies the types. 

84 # if coordinates are known; numpy.ndarray (size natom-by-3) 

85 if hasattr(data, "atomcoords"): 

86 attributes["atcoords"] = numpy.asanyarray(data.atomcoords)[-1] 

87 # if atomic numbers are known; numpy.ndarray (size natom) 

88 if hasattr(data, "atomnos"): 

89 attributes["atnums"] = numpy.asanyarray(data.atomnos) 

90 # if orbital coefficients known; iodata.orbitals.MolecularOrbitals 

91 if hasattr(data, "mocoeffs"): 

92 # First build a dictionary of inputs that will be passed to the constructor of 

93 # horton3's MolecularOrbitals object. 

94 moattr = { 

95 "kind": "restricted", 

96 "norba": data.nbasis, 

97 "norbb": None, 

98 # In horton 3, occupation in MOs are represented as 1's (occupied) and 

99 # 0's (unoccupied). Beta orbitals follow directly after alpha orbitals, forming 

100 # 1D array. 

101 "occs": numpy.concatenate( 

102 numpy.ones(data.homos[0]), numpy.zeros(data.nbasis - data.homos[0]) 

103 ), 

104 "coeffs": data.mocoeffs[0], 

105 "energies": None, 

106 "irreps": None, 

107 } 

108 # and if unrestricted: 

109 if len(mocoeffs) == 2: 

110 moattr["kind"] = "unrestricted" 

111 moattr["norbb"] = data.nbasis 

112 moattr["coeffs"].append(data.mocoeffs[1].T) 

113 moattr["occs"].append( 

114 numpy.concatenate( 

115 numpy.ones(data.homos[1]), 

116 numpy.zeros(data.nbasis - data.homos[1]), 

117 ) 

118 ) 

119 # Then construct MolecularOrbitals object 

120 attributes["mo"] = MolecularOrbitals(**moattr) 

121 # if multiplicity known; float / should not be set when mocoeffs present 

122 # Refer to IOData code: 

123 # https://github.com/theochem/iodata/blob/b36513d162f99b57264005583701c6987037839c/iodata/iodata.py#L174 

124 if ( 

125 hasattr(data, "mult") 

126 and not hasattr(data, "mocoeffs") 

127 ): 

128 attributes["spinpol"] = data.mult - 1 # horton has 2S+1, iodata has 2S 

129 # if pseudopotentials exist; numpy.ndarray (size natom) 

130 if hasattr(data, "coreelectrons"): 

131 attributes["atcorenums"] = data.atomnos - numpy.asanyarray(data.coreelectrons) 

132 # if mulliken charges are known; dict of numpy.ndarrays (size natom) 

133 if hasattr(data, "atomcharges"): 

134 attributes["atcharges"] = data.atomcharges 

135 

136 return IOData(**attributes) # Pass collected attributes into IOData constructor

139def makecclib(iodat): 

140 """ Create cclib ccData object from horton IOData object """ 

141 

142 hortonver = check_horton() 

143 attributes = {} 

144 

145 if hortonver == 2: 

146 # For a few attributes, a simple renaming suffices 

147 renameAttrs = { 

148 "numbers": "atomnos", 

149 "ms2": "mult", 

150 "polar": "polarizability", 

151 } 

152 inputattrs = iodat.__dict__ 

153 

154 attributes = dict( 

155 (renameAttrs[oldKey], val) 

156 for (oldKey, val) in inputattrs.items() 

157 if (oldKey in renameAttrs) 

158 ) 

159 

160 # Rest of attributes need some manipulation in data structure. 

161 if hasattr(iodat, "coordinates"): 

162 # cclib parses the whole history of coordinates in the list, horton keeps the last one. 

163 attributes["atomcoords"] = [iodat.coordinates] 

164 if hasattr(iodat, "orb_alpha"): 

165 attributes["mocoeffs"] = [iodat.orb_alpha] 

166 if hasattr(iodat, "orb_beta"): 

167 attributes["mocoeffs"].append(iodat.orb_beta) 

168 if hasattr(iodat, "pseudo_numbers"): 

169 # cclib stores num of electrons screened out by pseudopotential 

170 # horton stores num of electrons after applying pseudopotential 

171 attributes["coreelectrons"] = iodat.numbers - iodat.pseudo_numbers 

172 if hasattr(iodat, "mulliken_charges"): 

173 attributes["atomcharges"] = {"mulliken": iodat.mulliken_charges} 

174 if hasattr(iodat, "npa_charges"): 

175 attributes["atomcharges"]["natural"] = iodat.npa_charges 

176 elif hasattr(iodat, "npa_charges"): 

177 attributes["atomcharges"] = {"natural": iodat.npa_charges} 

178 

179 elif hortonver == 3: 

180 # Horton 3 IOData class uses attr and does not have __dict__. 

181 # In horton 3, some attributes have a default value of None. 

182 # Therefore, second hasattr statement is needed for mo attribute. 

183 if hasattr(iodat, "atcoords"): 

184 # cclib parses the whole history of coordinates in the list, horton keeps the last one. 

185 attributes["atomcoords"] = [iodat.atcoords] 

186 if hasattr(iodat, "mo") and hasattr(iodat.mo, "norba"): 

187 # MO coefficient should be transposed to match the dimensions. 

188 attributes["mocoeffs"] = [iodat.mo.coeffs[: iodat.mo.norba].T] 

189 if iodat.mo.kind == "unrestricted": 

190 attributes["mocoeffs"].append(iodat.mo.coeffs[iodat.mo.norba :].T) 

191 if hasattr(iodat, "spinpol") and isinstance(iodat.spinpol, int): 

192 # IOData stores 2S, ccData stores 2S+1. 

193 attributes["mult"] = iodat.spinpol + 1 

194 if hasattr(iodat, "atnums"): 

195 attributes["atnums"] = numpy.asanyarray(iodat.atnums) 

196 if hasattr(iodat, "atcorenums") and isinstance(iodat.atnums, numpy.ndarray): 

197 # cclib stores num of electrons screened out by pseudopotential 

198 # horton stores num of electrons after applying pseudopotential 

199 attributes["coreelectrons"] = numpy.asanyarray(iodat.atnums) - numpy.asanyarray( 

200 iodat.atcorenums 

201 ) 

202 if hasattr(iodat, "atcharges"): 

203 attributes["atomcharges"] = iodat.atcharges 

204 

205 return ccData(attributes)
