59 def parse_geometry(self, lines): 

60 """Parse DALTON geometry lines into an atomcoords array.""" 

61 

62 coords = [] 

63 for lin in lines: 

64 

65 # Without symmetry there are simply four columns, and with symmetry 

66 # an extra label is printed after the atom type. 

67 cols = lin.split() 

68 if cols[1][0] == "_": 

69 xyz = cols[2:] 

70 else: 

71 xyz = cols[1:] 

72 

73 # The assumption is that DALTON always print in atomic units. 

74 xyz = [utils.convertor(float(x), 'bohr', 'Angstrom') for x in xyz] 

75 coords.append(xyz) 

76 

77 return coords

82 # Extract the version number and optionally the Git revision 

83 # number. 

84 # 

85 # Example strings that at least the major version is parsed from: 

86 #  

87 # This is output from DALTON 2013.2 

88 # 2013.4

101 # release Dalton2018.0 (2018)

103 # release Dalton2018.alpha (2018) 

104 # release Dalton2019.alpha (2018) 

105 if line[4:30] == "This is output from DALTON": 

106 rs = (r"from DALTON \(?(?:Release|release)?\s?(?:Dalton)?" 

107 r"(\d+\.?[\w\d\-]*)" 

108 r"(?:[\s,]\(?)?")

110 if match: 

111 package_version = match.groups()[0] 

112 self.metadata["package_version"] = package_version 

113 self.metadata["legacy_package_version"] = package_version 

114 # Don't add revision information to the main package version for now. 

115 if "Last Git revision" in line: 

116 revision = line.split()[4] 

117 package_version = self.metadata.get("package_version") 

118 if package_version:

122 # manually specified? See before_parsing().

124 self.basislibrary = False

127 # about optimization targets (geotargets) and possibly other things as well. 

128 # Notice how the number of criteria required to converge is set to 2 here, 

129 # but this parameter can (probably) be tweaked in the input. 

130 # 

131 # Chosen parameters for *OPTIMI : 

132 # ------------------------------- 

133 # 

134 # Default 1st order method will be used: BFGS update. 

135 # Optimization will be performed in redundant internal coordinates (by default). 

136 # Model Hessian will be used as initial Hessian. 

137 # The model Hessian parameters of Roland Lindh will be used. 

138 # 

139 # 

140 # Trust region method will be used to control step (default). 

141 # 

142 # Convergence threshold for gradient set to : 1.00D-04 

143 # Convergence threshold for energy set to : 1.00D-06 

144 # Convergence threshold for step set to : 1.00D-04 

145 # Number of convergence criteria set to : 2

150 self.geotargets = []

156 self.geotargets.append(target)

159 # This is probably the first place where atomic symmetry labels are printed, 

160 # somewhere afer the SYMGRP point group information section. We need to know 

161 # which atom is in which symmetry, since this influences how some things are 

162 # print later on. We can also get some generic attributes along the way. 

163 # 

164 # Isotopic Masses 

165 # --------------- 

166 #

168 # C _2 12.000000 

169 # C _1 12.000000 

170 # C _2 12.000000 

171 # ... 

172 # 

173 # Note that when there is no symmetry there are only two columns here. 

174 # 

175 # It is also a good idea to keep in mind that DALTON, with symmetry on, operates 

176 # in a specific point group, so symmetry atoms have no internal representation. 

177 # Therefore only atoms marked as "_1" or "#1" in other places are actually 

178 # represented in the model. The symmetry atoms (higher symmetry indices) are 

179 # generated on the fly when writing the output. We will save the symmetry indices 

180 # here for later use. 

181 # 

182 # Additional note: the symmetry labels are printed only for atoms that have 

183 # symmetry images... so assume "_1" if a label is missing. For example, there will 

184 # be no label for atoms on an axes, such as the oxygen in water in C2v: 

185 # 

186 # O 15.994915 

187 # H _1 1.007825 

188 # H _2 1.007825
