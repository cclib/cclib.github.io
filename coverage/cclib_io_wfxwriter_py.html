17# Number of orbitals of type key. 

18# There are 3 p type, 6 d type orbitals etc. 

19ORBITAL_COUNT = {'S':1, 'P':3, 'D':6, 'F':10, 'G':15, 'H':21} 

20 

21# Index of first orbital of type key in a list of orbitals. 

22# The first s orbital has index 1, first p orbital has index 2, and first d 

23# has index 5. 

24ORBITAL_INDICES = {'S': 1} 

25ORBITAL_NAMES = 'SPDFGH' 

26for idx, name in enumerate(ORBITAL_NAMES[1:], start=1): 

27 prev_orbital_name = ORBITAL_NAMES[idx - 1] 

28 prev_orbital_count = ORBITAL_COUNT[prev_orbital_name] 

29 prev_orbital_index = ORBITAL_INDICES[prev_orbital_name] 

30 ORBITAL_INDICES[name] = prev_orbital_count + prev_orbital_index 

31 

32PI_CUBE_INV = (2.0 / numpy.pi) ** 3 

33 

34# Float formatting template. 

35WFX_FIELD_FMT = '%22.11E' 

36 

37# Precomputed values for l+m+n to be used in MO normalization. 

38_L = dict( 

39 [(prim_type, 0) for prim_type in range(1, 2)] + # s 

40 [(prim_type, 1) for prim_type in range(2, 5)] + # p 

41 [(prim_type, 2) for prim_type in range(5, 11)] + # d 

42 [(prim_type, 3) for prim_type in range(11, 21)] + # f 

43 [(prim_type, 4) for prim_type in range(21, 36)] # g 

44) 

45 

46# Precomputed values for ((2l-1)!! * (2m-1)!! * (2n-1)!!). 

47_M = dict( 

48 [(L, 1) for L in range(1, 5)] + 

49 [(L, 9) for L in range(5, 8)] + 

50 [(L, 1) for L in range(8, 11)] + 

51 [(L, 225) for L in range(11, 14)] + 

52 [(L, 9) for L in range(14, 20)] + 

53 [(L, 1) for L in range(20, 21)] + 

54 [(L, 11025) for L in range(21, 24)] + 

55 [(L, 225) for L in range(24, 30)] + 

56 [(L, 81) for L in range(30, 33)] + 

57 [(L, 9) for L in range(33, 36)] 

58)

61def _section(section_name, section_data): 

62 """Add opening/closing section_name tags to data.""" 

63 opening_tag = ['<' + section_name + '>'] 

64 closing_tag = ['</' + section_name + '>'] 

65 

66 section = None 

67 if isinstance(section_data, list): 

68 section = opening_tag + section_data + closing_tag 

69 elif isinstance(section_data, str): 

70 section = opening_tag + (' ' + section_data).split('\n') + closing_tag 

71 elif isinstance(section_data, int) or isinstance(section_data, float): 

72 section = opening_tag + [' ' + str(section_data)] + closing_tag 

73 return section

76def _list_format(data, per_line, style=WFX_FIELD_FMT): 

77 """Format lists for pretty print.""" 

78 template = style * per_line 

79 leftover = len(data) % per_line 

80 # Template for last line. 

81 last_template = style * leftover 

82 

83 pretty_list = [template%tuple(data[i:i+per_line]) 

84 for i in range(0, len(data) - leftover, per_line)] 

85 if leftover: 

86 return pretty_list + [last_template%tuple(data[-1*leftover:])] 

87 return pretty_list

207 def _rearrange_modata(self, data): 

208 """Rearranges MO related data according the expected order of 

209 Cartesian gaussian primitive types in wfx format. 

210 cclib parses mocoeffs in the order they occur in output files. 

211 """ 

212 prim_types = self._get_prim_types() 

213 if isinstance(data, numpy.ndarray): 

214 data = data.tolist() 

215 

216 pos_yyx = [key for key, val in enumerate(prim_types) 

217 if val == 17] 

218 pos_yyz = [key for key, val in enumerate(prim_types) 

219 if val == 16] 

220 

221 if pos_yyx: 

222 for pos in pos_yyx: 

223 data.insert(pos-3, data.pop(pos)) 

224 if pos_yyz: 

225 for pos in pos_yyz: 

226 data.insert(pos+3, data.pop(pos + 1)) 

227 

228 return data

231 def _get_prim_types(self): 

232 """List of primitive types. 

233 Definition of the Cartesian Gaussian primitive types is as follows: 

234 1 S, 2 PX, 3 PY, 4 PZ, 5 DXX, 6 DYY, 7 DZZ, 8 DXY, 9 DXZ, 10 DYZ, 

235 11 FXXX, 12 FYYY, 13 FZZZ, 14 FXXY, 15 FXXZ, 16 FYYZ, 17 FXYY, 

236 18 FXZZ, 19 FYZZ, 20 FXYZ, 

237 21 GXXXX, 22 GYYYY, 23 GZZZZ, 24 GXXXY, 25 GXXXZ, 26 GXYYY, 

238 27 GYYYZ, 28 GXZZZ, 

239 29 GYZZZ, 30 GXXYY, 31 GXXZZ, 32 GYYZZ, 33 GXXYZ, 34 GXYYZ, 

240 35 GXYZZ, 

241 36 HZZZZZ, 37 HYZZZZ, 38 HYYZZZ, 39 HYYYZZ, 

242 40 HYYYYZ, 41 HYYYYY, 42 HXZZZZ, 43 HXYZZZ, 44 HXYYZZ, 

243 45 HXYYYZ, 46 HXYYYY, 47 HXXZZZ, 48 HXXYZZ, 49 HXXYYZ, 

244 50 HXXYYY, 51 HXXXZZ, 52 HXXXYZ, 53 HXXXYY, 54 HXXXXZ, 55 HXXXXY, 

245 56 HXXXXX 

246 Spherical basis are not currently supported by the writer. 

247 """ 

248 prim_types = [] 

249 for atom in self.ccdata.gbasis: 

250 for prims in atom: 

251 prim_orb = [] 

252 for i in range(ORBITAL_COUNT[prims[0]]): 

253 prim_orb += [(ORBITAL_INDICES[prims[0]] + i)]\ 

254 * len(prims[1]) 

255 prim_types += prim_orb 

256 return prim_types

276 def _mo_occup_nos(self): 

277 """Section: Molecular Orbital Occupation Numbers.""" 

278 occup = [] 

279 electrons = self._no_electrons() 

280 alpha = self._no_alpha_electrons() 

281 beta = self._no_beta_electrons() 

282 if len(self.ccdata.homos) == 1: 

283 occup += [WFX_FIELD_FMT % (2)] * int(electrons / 2) +\ 

284 [WFX_FIELD_FMT % (1)] * (electrons % 2) 

285 else: 

286 occup += [WFX_FIELD_FMT % (1)] * + alpha +\ 

287 [WFX_FIELD_FMT % (1)] * beta 

288 return occup

291 """Section: Molecular Orbital Energies.""" 

292 mo_energies = [] 

293 alpha_elctrons = self._no_alpha_electrons() 

294 beta_electrons = self._no_beta_electrons()

305 """Section: Molecular Orbital Spin Types.""" 

306 spin_types = [] 

307 electrons = self._no_electrons() 

308 alpha = self._no_alpha_electrons() 

309 beta = self._no_beta_electrons()

318 def _normalize(self, prim_type, alpha=1.0): 

319 """Normalization factor for Cartesian Gaussian Functions. 

320 

321 N**4 = (2/pi)**3 * 2**(l+m+n) * alpha**(3 + 2(l+m+n)) / 

322 ((2l-1)!! * (2m-1)!! * (2n-1)!!)**2 

323 = (2/pi)**3 * 2**(L) * alpha**(3 + 2L) / 

324 M**2, 

325 L = l+m+n, 

326 M = ((2l-1)!! * (2m-1)!! * (2n-1)!!) 

327 """ 

328 L = _L[prim_type] 

329 M = _M[prim_type] 

330 norm_four = PI_CUBE_INV * 2**(4*L) * alpha**(3+2*L) / M 

331 norm = numpy.power(norm_four, 1/4.0) 

332 return norm

334 def _rearrange_mocoeffs(self, mocoeffs): 

335 """Rearrange cartesian F functions in mocoeffs. 

336 Expected order: 

337 xxx, yyy, zzz, xyy, xxy, xxz, xzz, yzz, yyz, xyz 

338 cclib's order for GAMESS: 

339 XXX, YYY, ZZZ, XXY, XXZ, YYX, YYZ, ZZX, ZZY, XYZ 

340 """ 

341 

342 aonames = self.ccdata.aonames 

343 mocoeffs = mocoeffs.tolist() 

344 

345 pos_yyx = [key for key, val in enumerate(aonames) 

346 if '_YYX' in val] 

347 pos_yyz = [key for key, val in enumerate(aonames) 

348 if '_YYZ' in val] 

349 

350 if pos_yyx: 

351 for pos in pos_yyx: 

352 mocoeffs.insert(pos-2, mocoeffs.pop(pos)) 

353 if pos_yyz: 

354 for pos in pos_yyz: 

355 mocoeffs.insert(pos+2, mocoeffs.pop(pos)) 

356 

357 return mocoeffs

359 def _norm_mat(self): 

360 """Calculate normalization matrix for normalizing MOcoeffs.""" 

361 alpha = [] 

362 prim_coeff = [] 

363 mo_count = [] 

364 prim_type = self._get_prim_types() 

365 

366 for atom in self.ccdata.gbasis: 

367 for prims in atom: 

368 prim_orb = [] 

369 mo_count += [len(prims[1])] * ORBITAL_COUNT[prims[0]] 

370 for i in range(ORBITAL_COUNT[prims[0]]): 

371 norb = ORBITAL_INDICES[prims[0]] 

372 prim_orb += [norb + i] 

373 alpha += [prim[0] for prim in prims[1]] 

374 prim_coeff += [prim[1] for prim in prims[1]] 

375 

376 # GAMESS specific reordering. 

377 if self.ccdata.metadata['package'] == 'GAMESS': 

378 prim_type = self._rearrange_modata(self._get_prim_types()) 

379 alpha = self._rearrange_modata(alpha) 

380 prim_coeff = self._rearrange_modata(prim_coeff) 

381 

382 norm_mat = [self._normalize(prim_type[i], alpha[i]) * prim_coeff[i] 

383 for i in range(len(prim_coeff))] 

384 

385 return (norm_mat, mo_count, prim_coeff)

393 def _prim_mocoeff(self, mo_count): 

394 """Return primitve mocoeffs array.""" 

395 prim_mocoeff = [] 

396 

397 for i in range(self.ccdata.mult): 

398 for j in range(self._nmos()): 

399 mocoeffs = self.ccdata.mocoeffs[i][j] 

400 if self.ccdata.metadata['package'] == 'GAMESS': 

401 mocoeffs = self._rearrange_mocoeffs(self.ccdata.mocoeffs[i][j]) 

402 for k, mocoeff in enumerate(mocoeffs): 

403 prim_mocoeff += [mocoeff] * mo_count[k] 

404 

405 return prim_mocoeff

407 def _normalized_mocoeffs(self): 

408 """Raw-Primitive Expansion coefficients for each normalized MO.""" 

409 # Normalization Matrix. 

410 norm_mat, mo_count, prim_coeff = self._norm_mat() 

411 

412 prim_mocoeff = self._prim_mocoeff(mo_count) 

413 

414 norm_mocoeffs = [] 

415 for mo_num in range(self._nmos()): 

416 norm_mocoeffs.append([norm_mat[i] * 

417 prim_mocoeff[i + mo_num * len(prim_coeff)] 

418 for i in range(len(prim_coeff))]) 

419 

420 return norm_mocoeffs

434 def _energy(self): 

435 """Section: Energy = T + Vne + Vee + Vnn. 

436 The total energy of the molecule. 

437 HF and KSDFT: SCF energy (scfenergies), 

438 MP2 : MP2 total energy (mpenergies), 

439 CCSD : CCSD total energy (ccenergies). 

440 """ 

441 energy = 0 

442 if hasattr(self.ccdata, 'ccenergies'): 

443 energy = self.ccdata.ccenergies[-1] 

444 elif hasattr(self.ccdata, 'mpenergies'): 

445 energy = self.ccdata.mpenergies[-1][-1] 

446 elif hasattr(self.ccdata, 'scfenergies'): 

447 energy = self.ccdata.scfenergies[-1] 

448 else: 

449 raise filewriter.MissingAttributeError( 

450 'scfenergies/mpenergies/ccenergies') 

451 return WFX_FIELD_FMT % (utils.convertor(energy, 'eV', 'hartree'))
