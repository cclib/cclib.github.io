62 def normalisesym(self, label): 

63 """Normalise the symmetries used by GAMESS. 

64 

65 To normalise, two rules need to be applied: 

66 (1) Occurences of U/G in the 2/3 position of the label 

67 must be lower-cased 

68 (2) Two single quotation marks must be replaced by a double 

69 """ 

70 

71 if label[1:] == "''": 

72 end = '"' 

73 else: 

74 end = label[1:].replace("U", "u").replace("G", "g") 

75 return label[0] + end
