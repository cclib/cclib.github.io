86 def extract(self, inputfile, line): 

87 """Extract information from the file object inputfile.""" 

88 

89 # Extract the package version. 

90 if "For non-commercial use only" in line: 

91 # Ignore the platorm information for now (the last character). 

92 self.metadata["package_version"] = line.split()[8][:-1] 

93 # Use the year as the legacy (short) package version. 

94 self.skip_lines( 

95 inputfile, ["Stewart Computational Chemistry", "s", "s", "s", "s"] 

96 ) 

97 self.metadata["legacy_package_version"] = next(inputfile).split()[1][5:] 

98 

99 # Extract the atomic numbers and coordinates from the optimized geometry 

100 # note that cartesian coordinates section occurs multiple times in the file, and we want to end up using the last instance 

101 # also, note that the section labeled cartesian coordinates doesn't have as many decimal places as the one used here 

102 # Example 1 (not used): 

103 # CARTESIAN COORDINATES 

104 # 

105 # NO. ATOM X Y Z 

106 # 

107 # 1 O 4.7928 -0.8461 0.3641 

108 # 2 O 5.8977 -0.3171 0.0092 

109 # ... 

110 # Example 2 (used): 

111 # ATOM CHEMICAL X Y Z 

112 # NUMBER SYMBOL (ANGSTROMS) (ANGSTROMS) (ANGSTROMS) 

113 # 

114 # 1 O 4.79280259 * -0.84610232 * 0.36409474 * 

115 # 2 O 5.89768035 * -0.31706418 * 0.00917035 * 

116 # ... etc. 

117 if line.split() == ["NUMBER", "SYMBOL", "(ANGSTROMS)", "(ANGSTROMS)", "(ANGSTROMS)"]: 

118 

119 self.updateprogress(inputfile, "Attributes", self.cupdate) 

120 

121 self.inputcoords = [] 

122 self.inputatoms = [] 

123 

124 blankline = inputfile.next() 

125 

126 atomcoords = [] 

127 line = inputfile.next() 

128 while len(line.split()) > 6: 

129 # MOPAC Version 14.019L 64BITS suddenly appends this block with 

130 # "CARTESIAN COORDINATES" block with no blank line. 

131 tokens = line.split() 

132 self.inputatoms.append(symbol2int(tokens[1])) 

133 xc = float(tokens[2]) 

134 yc = float(tokens[4]) 

135 zc = float(tokens[6]) 

136 atomcoords.append([xc, yc, zc]) 

137 line = inputfile.next() 

138 

139 self.inputcoords.append(atomcoords) 

140 

141 if not hasattr(self, "natom"): 

142 self.atomnos = numpy.array(self.inputatoms, 'i') 

143 self.natom = len(self.atomnos) 

144 

145 if 'CHARGE ON SYSTEM =' in line: 

146 charge = int(line.split()[5]) 

147 self.set_attribute('charge', charge) 

148 

149 if 'SPIN STATE DEFINED' in line: 

150 # find the multiplicity from the line token (SINGLET, DOUBLET, TRIPLET, etc) 

151 mult = self.spinstate[line.split()[1]] 

152 self.set_attribute('mult', mult) 

153 

154 # Read energy (in kcal/mol, converted to eV) 

155 # 

156 # FINAL HEAT OF FORMATION = -333.88606 KCAL = -1396.97927 KJ 

157 if 'FINAL HEAT OF FORMATION =' in line: 

158 if not hasattr(self, "scfenergies"): 

159 self.scfenergies = [] 

160 self.scfenergies.append(utils.convertor(utils.float(line.split()[5]), "kcal/mol", "eV")) 

161 

162 # Molecular mass parsing (units will be amu) 

163 # 

164 # MOLECULAR WEIGHT == 130.1890 

165 if line[0:35] == ' MOLECULAR WEIGHT =': 

166 self.molmass = utils.float(line.split()[3]) 

167 

168 #rotational constants 

169 #Example: 

170 # ROTATIONAL CONSTANTS IN CM(-1) 

171 # 

172 # A = 0.01757641 B = 0.00739763 C = 0.00712013 

173 # could also read in moment of inertia, but this should just differ by a constant: rot cons= h/(8*Pi^2*I) 

174 # note that the last occurence of this in the thermochemistry section has reduced precision, 

175 # so we will want to use the 2nd to last instance 

176 if line[0:40] == ' ROTATIONAL CONSTANTS IN CM(-1)': 

177 blankline = inputfile.next() 

178 rotinfo = inputfile.next() 

179 if not hasattr(self, "rotcons"): 

180 self.rotcons = [] 

181 broken = rotinfo.split() 

182 # leave the rotational constants in Hz 

183 a = float(broken[2]) 

184 b = float(broken[5]) 

185 c = float(broken[8]) 

186 self.rotcons.append([a, b, c]) 

187 

188 # Start of the IR/Raman frequency section. 

189 # Example: 

190 # VIBRATION 1 1A ATOM PAIR ENERGY CONTRIBUTION RADIAL 

191 # FREQ. 15.08 C 12 -- C 16 +7.9% (999.0%) 0.0% 

192 # T-DIPOLE 0.2028 C 16 -- H 34 +5.8% (999.0%) 28.0% 

193 # TRAVEL 0.0240 C 16 -- H 32 +5.6% (999.0%) 35.0% 

194 # RED. MASS 1.7712 O 1 -- O 4 +5.2% (999.0%) 0.4% 

195 # EFF. MASS7752.8338 

196 # 

197 # VIBRATION 2 2A ATOM PAIR ENERGY CONTRIBUTION RADIAL 

198 # FREQ. 42.22 C 11 -- C 15 +9.0% (985.8%) 0.0% 

199 # T-DIPOLE 0.1675 C 15 -- H 31 +6.6% (843.6%) 3.3% 

200 # TRAVEL 0.0359 C 15 -- H 29 +6.0% (802.8%) 24.5% 

201 # RED. MASS 1.7417 C 13 -- C 17 +5.8% (792.7%) 0.0% 

202 # EFF. MASS1242.2114 

203 if line[1:10] == 'VIBRATION': 

204 self.updateprogress(inputfile, "Frequency Information", self.fupdate) 

205 

206 # get the vib symmetry 

207 if len(line.split()) >= 3: 

208 sym = line.split()[2] 

209 if not hasattr(self, 'vibsyms'): 

210 self.vibsyms = [] 

211 self.vibsyms.append(sym) 

212 

213 line = inputfile.next() 

214 if 'FREQ' in line: 

215 if not hasattr(self, 'vibfreqs'): 

216 self.vibfreqs = [] 

217 freq = float(line.split()[1]) 

218 self.vibfreqs.append(freq) 

219 

220 line = inputfile.next() 

221 if 'T-DIPOLE' in line: 

222 if not hasattr(self, 'vibirs'): 

223 self.vibirs = [] 

224 tdipole = float(line.split()[1]) 

225 # transform to km/mol 

226 self.vibirs.append(math.sqrt(tdipole)) 

227 

228 # Orbital eigenvalues, e.g. 

229 # ALPHA EIGENVALUES 

230 # BETA EIGENVALUES 

231 # or just "EIGENVALUES" for closed-shell 

232 if 'EIGENVALUES' in line: 

233 if not hasattr(self, 'moenergies'): 

234 self.moenergies = [] # list of arrays 

235 

236 energies = [] 

237 line = inputfile.next() 

238 while len(line.split()) > 0: 

239 energies.extend([float(i) for i in line.split()]) 

240 line = inputfile.next() 

241 self.moenergies.append(energies) 

242 

243 # todo: 

244 # Partial charges and dipole moments 

245 # Example: 

246 # NET ATOMIC CHARGES 

247 

248 if line[:16] == '== MOPAC DONE ==': 

249 self.metadata['success'] = True
