38 def normalisesym(self, label): 

39 """Use standard symmetry labels instead of Gaussian labels. 

40 

41 To normalise: 

42 (1) If label is one of [SG, PI, PHI, DLTA], replace by [sigma, pi, phi, delta] 

43 (2) replace any G or U by their lowercase equivalent 

44 """ 

45 # note: DLT must come after DLTA 

46 greek = [('SG', 'sigma'), ('PI', 'pi'), ('PHI', 'phi'), 

47 ('DLTA', 'delta'), ('DLT', 'delta')] 

48 for k, v in greek: 

49 if label.startswith(k): 

50 tmp = label[len(k):] 

51 label = v 

52 if tmp: 

53 label = v + "." + tmp 

54 

55 ans = label.replace("U", "u").replace("G", "g") 

56 return ans
