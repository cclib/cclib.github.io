47 def __init__(self, ccdata, jobfilename=None, indices=None, terse=False, 

48 *args, **kwargs): 

49 """Initialize the Writer object. 

50 

51 This should be called by a subclass in its own __init__ method. 

52 

53 Inputs: 

54 ccdata - An instance of ccData, parsed from a logfile. 

55 jobfilename - The filename of the parsed logfile. 

56 indices - One or more indices for extracting specific geometries/etc. (zero-based) 

57 terse - Whether to print the terse version of the output file - currently limited to cjson/json formats 

58 """ 

59 

60 self.ccdata = ccdata 

61 self.jobfilename = jobfilename 

62 self.indices = indices 

63 self.terse = terse 

64 self.ghost = kwargs.get("ghost") 

65 

66 self.pt = PeriodicTable() 

67 

68 self._check_required_attributes() 

69 

70 # Open Babel isn't necessarily present. 

71 if _has_openbabel: 

72 # Generate the Open Babel/Pybel representation of the molecule. 

73 # Used for calculating SMILES/InChI, formula, MW, etc. 

74 self.obmol, self.pbmol = self._make_openbabel_from_ccdata() 

75 self.bond_connectivities = self._make_bond_connectivity_from_openbabel(self.obmol)

90 def _check_required_attributes(self): 

91 """Check if required attributes are present in ccdata.""" 

92 missing = [x for x in self.required_attrs 

93 if not hasattr(self.ccdata, x)] 

94 if missing: 

95 missing = ' '.join(missing) 

96 raise MissingAttributeError( 

97 'Could not parse required attributes to write file: ' + missing)

99 def _make_openbabel_from_ccdata(self): 

100 """Create Open Babel and Pybel molecules from ccData.""" 

101 if not hasattr(self.ccdata, 'charge'): 

102 logging.warning("ccdata object does not have charge, setting to 0") 

103 _charge = 0 

104 else: 

105 _charge = self.ccdata.charge 

106 if not hasattr(self.ccdata, 'mult'): 

107 logging.warning("ccdata object does not have spin multiplicity, setting to 1") 

108 _mult = 1 

109 else: 

110 _mult = self.ccdata.mult 

111 obmol = makeopenbabel(self.ccdata.atomcoords, 

112 self.ccdata.atomnos, 

113 charge=_charge, 

114 mult=_mult) 

115 if self.jobfilename is not None: 

116 obmol.SetTitle(self.jobfilename) 

117 return (obmol, pb.Molecule(obmol))

119 def _make_bond_connectivity_from_openbabel(self, obmol): 

120 """Based upon the Open Babel/Pybel molecule, create a list of tuples 

121 to represent bonding information, where the three integers are 

122 the index of the starting atom, the index of the ending atom, 

123 and the bond order. 

124 """ 

125 bond_connectivities = [] 

126 for obbond in ob.OBMolBondIter(obmol): 

127 bond_connectivities.append((obbond.GetBeginAtom().GetIndex(), 

128 obbond.GetEndAtom().GetIndex(), 

129 obbond.GetBondOrder())) 

130 return bond_connectivities

132 def _fix_indices(self): 

133 """Clean up the index container type and remove zero-based indices to 

134 prevent duplicate structures and incorrect ordering when 

135 indices are later sorted. 

136 """ 

137 if not self.indices: 

138 self.indices = set() 

139 elif not isinstance(self.indices, Iterable): 

140 self.indices = set([self.indices]) 

141 # This is the most likely place to get the number of 

142 # geometries from. 

143 if hasattr(self.ccdata, 'atomcoords'): 

144 lencoords = len(self.ccdata.atomcoords) 

145 indices = set() 

146 for i in self.indices: 

147 if i < 0: 

148 i += lencoords 

149 indices.add(i) 

150 self.indices = indices 

151 return
