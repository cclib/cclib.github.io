73class FileWrapper(object): 

74 """Wrap a file-like object or stream with some custom tweaks""" 

75 

76 def __init__(self, source, pos=0): 

77 

78 self.src = source 

79 

80 # Most file-like objects have seek and tell methods, but streams returned 

81 # by urllib.urlopen in Python2 do not, which will raise an AttributeError 

82 # in this code. On the other hand, in Python3 these methods do exist since 

83 # urllib uses the stream class in the io library, but they raise a different 

84 # error, namely io.UnsupportedOperation. That is why it is hard to be more 

85 # specific with except block here. 

86 try: 

87 self.src.seek(0, 2) 

88 self.size = self.src.tell() 

89 self.src.seek(pos, 0) 

90 

91 except (AttributeError, IOError, io.UnsupportedOperation): 

92 # Stream returned by urllib should have size information. 

93 if hasattr(self.src, 'headers') and 'content-length' in self.src.headers: 

94 self.size = int(self.src.headers['content-length']) 

95 else: 

96 self.size = pos 

97 

98 # Assume the position is what was passed to the constructor. 

99 self.pos = pos 

100 

101 self.last_line = None 

102 

103 def next(self): 

104 line = next(self.src) 

105 self.pos += len(line) 

106 self.last_line = line 

107 return line 

108 

109 def __next__(self): 

110 return self.next() 

111 

112 def __iter__(self): 

113 return self 

114 

115 def close(self): 

116 self.src.close()

118 def seek(self, pos, ref): 

119 

120 # If we are seeking to end, we can emulate it usually. As explained above, 

121 # we cannot be too specific with the except clause due to differences 

122 # between Python2 and 3. Yet another reason to drop Python 2 soon! 

123 try: 

124 self.src.seek(pos, ref) 

125 except: 

126 if ref == 2: 

127 self.src.read() 

128 else: 

129 raise 

130 

131 if ref == 0: 

132 self.pos = pos 

133 if ref == 1: 

134 self.pos += pos 

135 if ref == 2 and hasattr(self, 'size'): 

136 self.pos = self.size

139def openlogfile(filename, object=None): 

140 """Return a file object given a filename or if object specified decompresses it 

141 if needed and wrap it up. 

142 

143 Given the filename or file object of a log file or a gzipped, zipped, or bzipped 

144 log file, this function returns a file-like object. 

145 

146 Given a list of filenames, this function returns a FileInput object, 

147 which can be used for seamless iteration without concatenation. 

148 """ 

149 

150 # If there is a single string argument given. 

151 if type(filename) in [str, str]: 

152 

153 extension = os.path.splitext(filename)[1] 

154 

155 if extension == ".gz": 

156 fileobject = myGzipFile(filename, "r", fileobj=object) 

157 

158 elif extension == ".zip": 

159 zip = zipfile.ZipFile(object, "r") if object else zipfile.ZipFile(filename, "r") 

160 assert len(zip.namelist()) == 1, "ERROR: Zip file contains more than 1 file" 

161 fileobject = io.StringIO(zip.read(zip.namelist()[0]).decode("ascii", "ignore")) 

162 

163 elif extension in ['.bz', '.bz2']: 

164 # Module 'bz2' is not always importable. 

165 assert bz2 is not None, "ERROR: module bz2 cannot be imported" 

166 fileobject = myBZ2File(object, "r") if object else myBZ2File(filename, "r") 

167 

168 else: 

169 # Assuming that object is text file encoded in utf-8 

170 fileobject = io.StringIO(object.decode('utf-8')) if object \ 

171 else FileWrapper(io.open(filename, "r", errors='ignore')) 

172 

173 return fileobject 

174 

175 elif hasattr(filename, "__iter__"): 

176 

177 # This is needed, because fileinput will assume stdin when filename is empty. 

178 if len(filename) == 0: 

179 return None 

180 

181 # Compression (gzip and bzip) is supported as of Python 2.5. 

182 if sys.version_info[0] >= 2 and sys.version_info[1] >= 5: 

183 fileobject = fileinput.input(filename, openhook=fileinput.hook_compressed) 

184 else: 

185 fileobject = myFileinputFile(filename) 

186 

187 return fileobject

199 def __init__(self, source, loglevel=logging.ERROR, logname="Log", 

200 logstream=sys.stderr, datatype=ccData_optdone_bool, **kwds): 

201 """Initialise the Logfile object. 

202 

203 This should be called by a subclass in its own __init__ method. 

204 

205 Inputs: 

206 source - a logfile, list of logfiles, or stream with at least a read method 

207 loglevel - integer corresponding to a log level from the logging module 

208 logname - name of the source logfile passed to this constructor 

209 logstream - where to output the logging information 

210 datatype - class to use for gathering data attributes 

211 """ 

212 

213 # Set the filename to source if it is a string or a list of strings, which are 

214 # assumed to be filenames. Otherwise, assume the source is a file-like object 

215 # if it has a read method, and we will try to use it like a stream. 

216 self.isfileinput = False 

217 if isinstance(source, str): 

218 self.filename = source 

219 self.isstream = False 

220 elif isinstance(source, list) and all([isinstance(s, str) for s in source]): 

221 self.filename = source 

222 self.isstream = False 

223 elif isinstance(source, fileinput.FileInput): 

224 self.filename = source 

225 self.isstream = False 

226 self.isfileinput = True 

227 elif hasattr(source, "read"): 

228 self.filename = "stream %s" % str(type(source)) 

229 self.isstream = True 

230 self.stream = source 

231 else: 

232 raise ValueError("Unexpected source type.") 

233 

234 # Set up the logger. 

235 # Note that calling logging.getLogger() with one name always returns the same instance. 

236 # Presently in cclib, all parser instances of the same class use the same logger, 

237 # which means that care needs to be taken not to duplicate handlers. 

238 self.loglevel = loglevel 

239 self.logname = logname 

240 self.logger = logging.getLogger('%s %s' % (self.logname, self.filename)) 

241 self.logger.setLevel(self.loglevel) 

242 if len(self.logger.handlers) == 0: 

243 handler = logging.StreamHandler(logstream) 

244 handler.setFormatter(logging.Formatter("[%(name)s %(levelname)s] %(message)s")) 

245 self.logger.addHandler(handler) 

246 

247 # Set up the metadata. 

248 if not hasattr(self, "metadata"): 

249 self.metadata = {} 

250 self.metadata["package"] = self.logname 

251 self.metadata["methods"] = [] 

252 # Indicate if the computation has completed successfully 

253 self.metadata['success'] = False 

254 

255 

256 # Periodic table of elements. 

257 self.table = utils.PeriodicTable() 

258 

259 # This is the class that will be used in the data object returned by parse(), and should 

260 # normally be ccData or a subclass of it. 

261 self.datatype = datatype 

262 

263 # Change the class used if we want optdone to be a list or if the 'future' option 

264 # is used, which might have more consequences in the future. 

265 optdone_as_list = kwds.get("optdone_as_list", False) or kwds.get("future", False) 

266 optdone_as_list = optdone_as_list if isinstance(optdone_as_list, bool) else False 

267 if optdone_as_list: 

268 self.datatype = ccData 

269 # Parsing of Natural Orbitals and Natural Spin Orbtials into one attribute 

270 self.unified_no_nso = kwds.get("future",False)

287 def parse(self, progress=None, fupdate=0.05, cupdate=0.002): 

288 """Parse the logfile, using the assumed extract method of the child.""" 

289 

290 # Check that the sub-class has an extract attribute, 

291 # that is callable with the proper number of arguemnts. 

292 if not hasattr(self, "extract"): 

293 raise AttributeError("Class %s has no extract() method." % self.__class__.__name__) 

294 if not callable(self.extract): 

295 raise AttributeError("Method %s._extract not callable." % self.__class__.__name__) 

296 if len(getargspec(self.extract)[0]) != 3: 

297 raise AttributeError("Method %s._extract takes wrong number of arguments." % self.__class__.__name__) 

298 

299 # Save the current list of attributes to keep after parsing. 

300 # The dict of self should be the same after parsing. 

301 _nodelete = list(set(self.__dict__.keys())) 

302 

303 # Initiate the FileInput object for the input files. 

304 # Remember that self.filename can be a list of files. 

305 if not self.isstream: 

306 if not self.isfileinput: 

307 inputfile = openlogfile(self.filename) 

308 else: 

309 inputfile = self.filename 

310 else: 

311 inputfile = FileWrapper(self.stream) 

312 

313 # Intialize self.progress 

314 is_compressed = isinstance(inputfile, myGzipFile) or isinstance(inputfile, myBZ2File) 

315 if progress and not (is_compressed): 

316 self.progress = progress 

317 self.progress.initialize(inputfile.size) 

318 self.progress.step = 0 

319 self.fupdate = fupdate 

320 self.cupdate = cupdate 

321 

322 # Maybe the sub-class has something to do before parsing. 

323 self.before_parsing() 

324 

325 # Loop over lines in the file object and call extract(). 

326 # This is where the actual parsing is done. 

327 for line in inputfile: 

328 self.updateprogress(inputfile, "Unsupported information", cupdate) 

329 

330 # This call should check if the line begins a section of extracted data. 

331 # If it does, it parses some lines and sets the relevant attributes (to self). 

332 # Any attributes can be freely set and used across calls, however only those 

333 # in data._attrlist will be moved to final data object that is returned. 

334 try: 

335 self.extract(inputfile, line) 

336 except StopIteration: 

337 self.logger.error("Unexpectedly encountered end of logfile.") 

338 break 

339 except Exception as e: 

340 self.logger.error("Encountered error when parsing.") 

341 self.logger.error("Last line read: %s" % inputfile.last_line) 

342 raise 

343 

344 # Close input file object. 

345 if not self.isstream: 

346 inputfile.close() 

347 

348 # Maybe the sub-class has something to do after parsing. 

349 self.after_parsing() 

350 

351 # If atomcoords were not parsed, but some input coordinates were ("inputcoords"). 

352 # This is originally from the Gaussian parser, a regression fix. 

353 if not hasattr(self, "atomcoords") and hasattr(self, "inputcoords"): 

354 self.atomcoords = numpy.array(self.inputcoords, 'd') 

355 

356 # Set nmo if not set already - to nbasis. 

357 if not hasattr(self, "nmo") and hasattr(self, "nbasis"): 

358 self.nmo = self.nbasis 

359 

360 # Create a default coreelectrons array, unless it's impossible 

361 # to determine. 

362 if not hasattr(self, "coreelectrons") and hasattr(self, "natom"): 

363 self.coreelectrons = numpy.zeros(self.natom, "i") 

364 if hasattr(self, "incorrect_coreelectrons"): 

365 self.__delattr__("coreelectrons") 

366 

367 # Create the data object we want to return. This is normally ccData, but can be changed 

368 # by passing the datatype argument to the constructor. All supported cclib attributes 

369 # are copied to this object, but beware that in order to be moved an attribute must be 

370 # included in the data._attrlist of ccData (or whatever else). 

371 # There is the possibility of passing assitional argument via self.data_args, but 

372 # we use this sparingly in cases where we want to limit the API with options, etc. 

373 data = self.datatype(attributes=self.__dict__) 

374 

375 # Now make sure that the cclib attributes in the data object are all the correct type, 

376 # including arrays and lists of arrays. 

377 data.arrayify() 

378 

379 # Delete all temporary attributes (including cclib attributes). 

380 # All attributes should have been moved to a data object, which will be returned. 

381 for attr in list(self.__dict__.keys()): 

382 if not attr in _nodelete: 

383 self.__delattr__(attr) 

384 

385 # Perform final checks on values of attributes. 

386 data.check_values(logger=self.logger) 

387 

388 # Update self.progress as done. 

389 if hasattr(self, "progress"): 

390 self.progress.update(inputfile.size, "Done") 

391 

392 return data

469 def _assign_coreelectrons_to_element(self, element, ncore, 

470 ncore_is_total_count=False): 

471 """Assign core electrons to all instances of the element. 

472 

473 It's usually reasonable to do this for all atoms of a given element, 

474 because mixed usage isn't normally allowed within elements. 

475 

476 Parameters 

477 ---------- 

478 element: str 

479 the chemical element to set coreelectrons for 

480 ncore: int 

481 the number of core electrons 

482 ncore_is_total_count: bool 

483 whether the ncore argument is the total count, in which case it is 

484 divided by the number of atoms of this element 

485 """ 

486 atomsymbols = [self.table.element[atomno] for atomno in self.atomnos] 

487 indices = [i for i, el in enumerate(atomsymbols) if el == element] 

488 if ncore_is_total_count: 

489 ncore = ncore // len(indices) 

490 

491 if not hasattr(self, 'coreelectrons'): 

492 self.coreelectrons = numpy.zeros(self.natom, 'i') 

493 self.coreelectrons[indices] = ncore

495 def skip_lines(self, inputfile, sequence): 

496 """Read trivial line types and check they are what they are supposed to be. 

497 

498 This function will read len(sequence) lines and do certain checks on them, 

499 when the elements of sequence have the appropriate values. Currently the 

500 following elements trigger checks: 

501 'blank' or 'b' - the line should be blank 

502 'dashes' or 'd' - the line should contain only dashes (or spaces) 

503 'equals' or 'e' - the line should contain only equal signs (or spaces) 

504 'stars' or 's' - the line should contain only stars (or spaces) 

505 """ 

506 

507 expected_characters = { 

508 '-': ['dashes', 'd'], 

509 '=': ['equals', 'e'], 

510 '*': ['stars', 's'], 

511 } 

512 

513 lines = [] 

514 for expected in sequence: 

515 

516 # Read the line we want to skip. 

517 line = next(inputfile) 

518 

519 # Blank lines are perhaps the most common thing we want to check for. 

520 if expected in ["blank", "b"]: 

521 try: 

522 assert line.strip() == "" 

523 except AssertionError: 

524 frame, fname, lno, funcname, funcline, index = inspect.getouterframes(inspect.currentframe())[1] 

525 parser = fname.split('/')[-1] 

526 msg = "In %s, line %i, line not blank as expected: %s" % (parser, lno, line.strip()) 

527 self.logger.warning(msg) 

528 

529 # All cases of heterogeneous lines can be dealt with by the same code. 

530 for character, keys in expected_characters.items(): 

531 if expected in keys: 

532 try: 

533 assert all([c == character for c in line.strip() if c != ' ']) 

534 except AssertionError: 

535 frame, fname, lno, funcname, funcline, index = inspect.getouterframes(inspect.currentframe())[1] 

536 parser = fname.split('/')[-1] 

537 msg = "In %s, line %i, line not all %s as expected: %s" % (parser, lno, keys[0], line.strip()) 

538 self.logger.warning(msg) 

539 continue 

540 

541 # Save the skipped line, and we will return the whole list. 

542 lines.append(line) 

543 

544 return lines
